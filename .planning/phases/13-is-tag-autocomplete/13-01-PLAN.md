  <action>
Create a new Svelte 5 component at `src/lib/components/cards/IsTagAutocomplete.svelte`.

Use runes syntax throughout ($state, $derived, $effect, $props).

```svelte
<script lang="ts">
  import { ORACLE_TAGS, ORACLE_TAG_LABELS } from '$lib/data/oracle-tags'

  let { query, onselect }: { query: string; onselect: (newQuery: string) => void } = $props()

  let dropdownEl: HTMLDivElement | null = $state(null)
  let dismissedAtQuery = $state('')

  // Extract the partial tag text after the last is: token (if the query ends with one)
  const partialText = $derived((/is:(\S*)$/i.exec(query) ?? [])[1]?.toLowerCase() ?? null)

  // All known tag keys in alphabetical order
  const allTags = Object.keys(ORACLE_TAGS).sort()

  // Filter to tags whose key starts with the partial text
  const filteredTags = $derived(
    partialText !== null ? allTags.filter((t) => t.startsWith(partialText)) : []
  )

  // Show when: partial text present, tags available, and user hasn't dismissed this exact query
  const visible = $derived(
    partialText !== null && filteredTags.length > 0 && query !== dismissedAtQuery
  )

  function handleSelect(tag: string) {
    // Replace trailing is:partial with is:tag followed by a space (space acts as natural dismiss)
    onselect(query.replace(/is:\S*$/i, 'is:' + tag + ' '))
  }

  $effect(() => {
    function handleMousedown(e: MouseEvent) {
      if (dropdownEl && !dropdownEl.contains(e.target as Node)) {
        dismissedAtQuery = query
      }
    }
    document.addEventListener('mousedown', handleMousedown)
    return () => document.removeEventListener('mousedown', handleMousedown)
  })

  // Reset dismiss state when query changes away from the dismissed value
  $effect(() => {
    if (query !== dismissedAtQuery) {
      // Nothing — just let visible re-derive naturally
    }
  })
</script>

{#if visible}
  <div
    bind:this={dropdownEl}
    class="absolute left-0 right-0 top-full z-50 mt-1 max-h-64 overflow-y-auto rounded-md border bg-popover shadow-md"
  >
    {#each filteredTags as tag (tag)}
      <button
        type="button"
        class="flex w-full items-center justify-between px-3 py-2 text-sm hover:bg-accent hover:text-accent-foreground"
        onmousedown={(e) => {
          // Use mousedown (before blur) to register the selection
          e.preventDefault()
          handleSelect(tag)
        }}
      >
        <span>{ORACLE_TAG_LABELS[tag] ?? tag}</span>
        <span class="ml-4 text-muted-foreground">is:{tag}</span>
      </button>
    {/each}
  </div>
{/if}
```

Key design choices:

- `onmousedown` with `e.preventDefault()` instead of `onclick` — prevents the input from losing focus before the selection registers
- `partialText` regex `/is:(\S*)$/i` — matches only when query ends with an is: token (ongoing completion), not mid-query tokens
- Space after tag in `handleSelect` — appending a space means the query no longer ends with an `is:` token, so the dropdown disappears naturally on next render
- Click-outside via `mousedown` on document sets `dismissedAtQuery = query` — dropdown hides for this exact query string
